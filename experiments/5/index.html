  function createFormComponent(name = '') {
    return {
      text: '',
      clock: new Date().toLocaleTimeString(),
      get time() {
        return `The time is ${this.clock} ${name ? `of ${name}` : ''} `;
      },
      makeClock: function () {
        setInterval(() => {
          this.clock = new Date().toLocaleTimeString();
        }, 1000);
      }
    };
  };
